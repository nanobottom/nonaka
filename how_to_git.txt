# gitのcloneを作成する
1. ディレクトリを作成する
$ mkdir test
2. ディレクトリに移動する
$ cd test
3. クローン
git clone <clone URL>
clone URL : https://github.com/nanobottom/nonaka.git
# GitHubに変更を反映する/ファイルを追加する
1. ステージング
$ git add ファイル名
2. コミット
$ git commit -m "メッセージ"
3.プッシュ
$ git push

# 現在のブランチを変更する
$ git branch

# ブランチを変更する
$ git checkout ブランチ名

# ログを確認する
$ git log -n 数値

# 状態を確認する（コミットしてないファイルなど）
$ git status

# GitHubから要らないファイルを削除する
1. ローカルファイルを削除
$ git rm ファイル名
$ git rm -r ディレクトリ名
2. ステージング
$ git add .
3. コミット
$ git commit -m "メッセージ"
4. プッシュ
$ git push
